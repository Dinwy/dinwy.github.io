public int[] Solution(int[] digits)
{
    var reminder = 1;
    for (int i = digits.Length - 1; i >= 0; i--)
    {
        digits[i] += reminder;

        if (digits[i] >= 10)
        {
            digits[i] = 0;
            reminder = 1;
            continue;
        }
        reminder = 0;

        break;
    }

    if (reminder != 0) {
        var arr = new int[digits.Length + 1];
        arr[0] = reminder;
        Array.Copy(digits, 0, arr, 1, digits.Length);
        return arr;
    }
    return digits;
}
